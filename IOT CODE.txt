int a=2;
int d=3;

void setup()
{
  Serial.begin(9600);
  pinMode(a,OUTPUT);
  pinMode(d,INPUT);
  pinMode(12,OUTPUT);
}

void loop()
{
  //ultrasonic sensor
  digitalWrite(a,LOW);
  digitalWrite(a,HIGH);
  delayMicroseconds(10);
  digitalWrite(a,LOW);
  float du=pulseIn(d,HIGH);
  float dis=(du*0.0343)/2;
  Serial.print("Distance is: ");
  Serial.println(dis);
  
    //LED ON
  if(dis>=100)
  {
    digitalWrite(8,HIGH);
    digitalWrite(7,HIGH);
  }
  
  //Buzzer For ultrasonic Sensor
  if(dis>=100)
  {
  for(int i=0; i<=30000; i=i+10)
  {
  tone(12,i);
  delay(1000);
  noTone(12);
  delay(1000);
  }
  }
  
  
  
  
    //Temperate Sensor
  double b= analogRead(A0);
  double a=(((a/1024)*5)-0.5)*100;
  Serial.print("Temp Value: ");
  Serial.println(a);
  delay(1000);
  
  
  //LED ON
  if(a>=100)
  {
    digitalWrite(8,HIGH);
    digitalWrite(7,HIGH);
  }
  
  //Buzzer for Temperature Sensor
  if(a>=100)
  {
  for(int i=0; i<=30000; i=i+10)
  {
  tone(12,i);
  delay(1000);
  noTone(12);
  delay(1000);
  }
  }

   //LED OFF
  if(a<100)
  {
    digitalWrite(8,LOW);
    digitalWrite(7,LOW);
  }
}
